%%%%%%%%%%%%%%%%%%%%%%%%%
% Default font families %
%%%%%%%%%%%%%%%%%%%%%%%%%

% Serif (Roman)
\textrm{Text} % command
{\rmfamily Text} % switch

% Sans Serif
\textsf{Text}
{\sffamily Text}

% Typewriter (Monospace)
\texttt{Text}
{\ttfamily Text}


%%%%%%%%%%%%%%
% Font style %
%%%%%%%%%%%%%%

% Medium
\textmd{Text}
{\mdseries Text}

% Bold
\textbf{Text}
{\bfseries Text}

% Upright
\textup{Text}
{\upshape Text}

% Italic
\textit{Text}
{\itshape Text}

% Slanted
\textsl{Text}
{\slshape Text}

% Small Caps
\textsc{Text}
{\scshape Text}

% Bold + italic
\textit{\textbf{Fett-Kursiv}} 
{\bfseries\itshape Alles in dieser Gruppe ist fett-kursiv}


%%%%%%%%%%%%%%
% Decoration %
%%%%%%%%%%%%%%

% Underline
\underline{Text} % command
{\protect\underline Text} % switch-like usage (use with care)

% Strikethrough (requires \usepackage{ulem})
\sout{Text}

% Wavy underline (requires \usepackage{ulem})
\uwave{Text}

% Double underline (requires \usepackage{ulem})
\uuline{Text}

% Dashed underline (requires \usepackage{ulem})
\dashuline{Text}

% Dotted underline (requires \usepackage{ulem})
\dotuline{Text}


%%%%%%%%%%%%%%%%%%
% Text alignment %
%%%%%%%%%%%%%%%%%%

% Left-aligned (Flush left)
\begin{flushleft} Text \end{flushleft} % environment
{\raggedright Text} % switch

% Centered
\begin{center} Text \end{center} % environment
{\centering Text} % switch

% Right-aligned (Flush right)
\begin{flushright} Text \end{flushright} % environment
{\raggedleft Text} % switch


%%%%%%%%%%%%%
% Font size %
%%%%%%%%%%%%%

% Tiny
{\tiny Text} % switch

% Scriptsize
{\scriptsize Text}

% Footnotesize
{\footnotesize Text}

% Small
{\small Text}

% Normalsize
{\normalsize Text}

% Large
{\large Text}

% Large (bigger)
{\Large Text}

% LARGE (even bigger)
{\LARGE Text}

% Huge
{\huge Text}

% Huge (biggest)
{\Huge Text}
