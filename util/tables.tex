% Syntax: \hlinestable{Label}{Caption}{Column definition}{Content}
\NewDocumentCommand{\hlinestable}{m m m +m}{
    \begin{table}[h]
        \centering
        % Nutze tblr (statt talltblr), um Konflikte mit der Caption-Logik zu vermeiden
        \begin{tblr}{
            colspec = {#3},
            hline{1,2,Z} = {0.08em},
            row{1} = {font=\bfseries},
        }
            #4
        \end{tblr}
        \caption{#2}
        \label{#1}
    \end{table}
}
